00039 double ran2_dp(int idum) 
00040 {
00041   int IM1=2147483563,IM2=2147483399,IMM1=IM1-1, 
00042     IA1=40014,IA2=40692,IQ1=53668,IQ2=52774,IR1=12211,IR2=3791, 
00043     NTAB=32,NDIV=1+IMM1/NTAB;
00044 
00045   double AM=1./IM1,EPS=1.2e-7,RNMX=1.-EPS;
00046   int j,k;
00047   static int iv[32],iy,idum2 = 123456789; 
00048   // N.B. in C static variables are initialised to 0 by default.
00049 
00050   if (idum <= 0) {
00051     idum= (-idum>1 ? -idum : 1); // max(-idum,1);
00052     idum2=idum;
00053     for(j=NTAB+8;j>=1;j--) {
00054       k=idum/IQ1;
00055       idum=IA1*(idum-k*IQ1)-k*IR1;
00056       if (idum < 0) idum=idum+IM1;
00057       if (j < NTAB) iv[j-1]=idum;
00058     }
00059     iy=iv[0];
00060   }
00061   k=idum/IQ1;
00062   idum=IA1*(idum-k*IQ1)-k*IR1;
00063   if (idum < 0) idum=idum+IM1;
00064   k=idum2/IQ2;
00065   idum2=IA2*(idum2-k*IQ2)-k*IR2;
00066   if (idum2 < 0) idum2=idum2+IM2;
00067   j=1+iy/NDIV;
00068   iy=iv[j-1]-idum2;
00069   iv[j-1]=idum;
00070   if(iy < 1)iy=iy+IMM1;
00071   return (AM*iy < RNMX ? AM*iy : RNMX); // min(AM*iy,RNMX);
00072 
00073 }
